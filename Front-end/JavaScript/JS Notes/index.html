JavaScript is Single Threaded

JavaScript is a single-threaded language, meaning it can execute only one code block at a time. This implies that JavaScript cannot run multiple pieces of code simultaneously.

When making API calls or saving data to a database, if the response takes a long time, the JavaScript program cannot wait indefinitely. To handle such situations,

You can use setTimeout() to manage asynchronous operations. This process is handled by the browser, which is written in multithreaded languages like Java and C++.

Promises

The Promise object represents the eventual completion (or failure) of an asynchronous operation and its resulting value.

resolve, reject, then( ) & catch( ) 

        
            // This a dummy DB funcrion
            function saveToDB(data) {
                return new Promise((resolve, reject) => {
                    let internetSpeed = Math.floor(Math.random() * 2);
                    if (internetSpeed == 1) {
                        resolve("success : data was saved");
                    } else {
                        reject("failure : weak connectionl");
                    }
                });
            }
    
            saveToDB("Azim")
                .then(() => {
                    console.log("Promise was resolved!");
                })
                .catch(() => {
                    console.log("Promise was rejected!");
                });

Async Functions

An async function in JavaScript allows you to handle asynchronous operations using `await` to wait for promises, simplifying code flow.

await pauses the execution of its surrounding async function until the promise is settled (resolved or rejected) 

Async Function can be ushed in arrow function as well

Createing an Async Function: 

        
            async function azim() {
                return "Hello Azim"; // returns a promise
            }
    
            let azim2 = async () => { } // returns a promise
    
            // async alyays returns a promise

Using .then & .catch: 

        
            async function azim() {
                throw "404 page not found!";
                return "Hello Azim";
            }
    
            azim()
            .then((result) => {
                console.log("Promise Was resolved!");
                console.log("result : ", result);
            })
            .catch((err) => {
                console.log("promise was rejected with err : ", err);
            });
            
            // ERROR OUTPUT //
            // promise was rejected with err :  404 page not found!

            // RESOLVED OUTPUT //
            // Promise Was resolved!
            // result :  Hello Azim

            // Random Jokes API
            let url = 'https://v2.jokeapi.dev/joke/Any?safe-mode';

            fetch(url)
                .then((res) => {

Using Async and await 

        
            // Random Jokes API
            let url = "https://v2.jokeapi.dev/joke/Any?safe-mode";
    
            async function getJokes() {
                try {
                    let res = await fetch(url);
                    let data = await res.json();
                    console.log(data.setup);
                    console.log(data.delivery);
                } catch (err) {
                    console.log("ERROR >> ", err);
                }
            }
    
            getJokes();
